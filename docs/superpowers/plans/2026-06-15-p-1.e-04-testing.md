        "# Sample rubric\n\n"
        "| # | name | weight |\n"
        "|---|------|--------|\n"
        "| 1 | creativity | 60 |\n"
        "| 2 | consistency | 40 |\n\n"
        "## Kill Switches\n\n"
        "- total score = 0 on plagiarism\n"
    )
