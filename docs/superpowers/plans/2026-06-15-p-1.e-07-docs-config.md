**Design choice — explicit list vs auto-discovery**: The skill list in `master.json` is explicit (hardcoded paths) rather than auto-discovered via `skills/*/SKILL.md`. Trade-offs:

- **Explicit (chosen)**: Adding a skill requires a 2-line PR edit (add to `master.json`, add the skill directory). The CI freshness check catches omissions. Reviewers see the skill roster in the PR diff.
- **Auto-discovery (rejected)**: Generator globs `skills/*/SKILL.md` at runtime. Pro: zero maintenance. Con: a misnamed skill folder (e.g., `skills/shenbi_worldbuilding/` instead of `skills/shenbi-worldbuilding/`) silently appears in the manifest with a broken identifier; no PR-level review of skill roster changes; harder to mark skills as "disabled" or "internal-only".

If the project grows to 100+ skills and maintenance becomes painful, switch to auto-discovery + an explicit denylist. Tracked as future work, not P-1.E scope.

```python
"""Generate per-platform plugin manifests from master.json.

Single source of truth: plugins/master.json
Output: 4 platform manifests (.claude-plugin/, .codex-plugin/,
.cursor-plugin/, .opencode/).
"""

import json
import sys
from pathlib import Path
from typing import Any

import structlog

log = structlog.get_logger(__name__)

REPO_ROOT = Path(__file__).resolve().parents[3]
MASTER_PATH = REPO_ROOT / "plugins" / "master.json"


def load_master() -> dict[str, Any]:
    """Load master.json (structural validation is a future enhancement)."""
    if not MASTER_PATH.exists():
        raise FileNotFoundError(f"Master plugin file not found: {MASTER_PATH}")
    return json.loads(MASTER_PATH.read_text(encoding="utf-8"))


# Each generator builds its output dict/list in the platform's canonical key
# order and pulls platform-specific fields from config["fields"]. Keeping the
# order explicit makes generated output match the existing committed manifests
# (so `git diff` after regeneration shows only real changes).


def _common_header(master: dict) -> dict:
    """Fields shared by all JSON platforms, in canonical key order."""
    return {
        "name": master["name"],
        "version": master["version"],
        "description": master["description"],
        "author": master["author"],
    }


def gen_claude(master: dict, config: dict) -> dict:
    out = _common_header(master)
    out["skills"] = master["skills"]
    return out


def gen_codex(master: dict, config: dict) -> dict:
    fields = config.get("fields", {})
    out = _common_header(master)
    out["marketplace"] = fields["marketplace"]
    out["type"] = fields["type"]
    out["skills"] = master["skills"]
    return out


def gen_cursor(master: dict, config: dict) -> dict:
    fields = config.get("fields", {})
    out = _common_header(master)
    out["pluginRoot"] = fields["pluginRoot"]
    out["skills"] = master["skills"]
    out["hooks"] = fields["hooks"]
    return out


def gen_opencode(master: dict, config: dict) -> str:
    """OpenCode uses an ES module (export default) with single-quoted strings."""
    lines = [
        "export default {",
        f"  name: '{master['name']}',",
        f"  version: '{master['version']}',",
        f"  description: '{master['description']}',",
        f"  author: '{master['author']}',",
        "  skills: [",
    ]
    for skill in master["skills"]:
        lines.append(f"    '{skill}',")
    # Drop trailing comma on the final entry to match the committed file.
    lines[-1] = lines[-1].rstrip(",")
    lines.append("  ]")
    lines.append("};")
    return "\n".join(lines) + "\n"


GENERATORS = {
    "claude-code": ("json", gen_claude),
    "codex-cli": ("json", gen_codex),
    "cursor": ("json", gen_cursor),
    "opencode-js": ("js", gen_opencode),
}


def generate_all() -> int:
    """Generate all platform manifests from master.json."""
    master = load_master()

    for platform_name, config in master["platforms"].items():
        fmt = config["format"]
        output_path = REPO_ROOT / config["output"]
        output_path.parent.mkdir(parents=True, exist_ok=True)

        if fmt not in GENERATORS:
            log.error("unknown_format", format=fmt, platform=platform_name)
            return 1

        output_type, generator = GENERATORS[fmt]
        content = generator(master, config)

        if output_type == "json":
            output_path.write_text(
                json.dumps(content, indent=2, ensure_ascii=False), encoding="utf-8"
            )
        else:
            output_path.write_text(content, encoding="utf-8")

        log.info("generated", path=str(output_path.relative_to(REPO_ROOT)))

    return 0


def main() -> int:
    return generate_all()


if __name__ == "__main__":
    sys.exit(main())
```

```python
"""Verify documentation accuracy.

Scans markdown files for code-span references to file paths (e.g.,
`tests/validate-gate.py`) and verifies those files exist. Catches the
most common doc drift: stale path references.

Not a substitute for semantic review — only catches missing-file drift.
"""

import re
from pathlib import Path

import pytest

REPO_ROOT = Path(__file__).resolve().parents[2]

# docs/getting-started/* are intentionally excluded: that directory does not
# exist yet and is not P-1.E scope (it arrives with the P0 user-guide work).
# The test skips any listed file that is missing, so they can be re-added later.
DOCS_TO_CHECK = [
    "AGENTS.md",
    "CLAUDE.md",
    "GEMINI.md",
    "README.md",
    "CONTRIBUTING.md",
    "SECURITY.md",
    "CHANGELOG.md",
    "command-to-give.md",
    "goal-prompt.md",
    "docs/index.md",
]

# Paths referenced in docs that may not exist in main branch yet
# (e.g., referenced as "future" or in spec context). Add here with reason.
ALLOWED_MISSING: set[str] = {
    # Example: "docs/api/exceptions.md" might be referenced in nav but is a stub
}

# Match relative-path codespans: letters, digits, dots, slashes, hyphens, underscores.
# Reject commands-with-args ("bash tests/foo.sh"), template placeholders
# ("skills/<skill>/SKILL.md"), and URLs.
CODESPAN_PATTERN = re.compile(r"`([\w][\w./-]*\.\w+)`")
HISTORICAL_PATH_PATTERN = re.compile(r"tests/rounds/round-\d+")


def extract_paths_from_doc(doc_path: Path) -> list[Path]:
    """Extract code-span file path references from a markdown doc."""
    if not doc_path.exists():
        return []
    text = doc_path.read_text(encoding="utf-8")
    paths = []
    for match in CODESPAN_PATTERN.finditer(text):
        candidate = match.group(1)
        # Only consider paths with slashes or known extensions
        if "/" in candidate or candidate.endswith((
            ".py", ".md", ".yaml", ".yml", ".toml", ".json", ".sh",
        )):
            # Skip historical round references
            if HISTORICAL_PATH_PATTERN.search(candidate):
                continue
            paths.append(REPO_ROOT / candidate)
    return paths


@pytest.mark.parametrize("doc_relative", DOCS_TO_CHECK)
def test_doc_references_existing_files(doc_relative: str):
    """Each code-span file reference in docs must point to an existing file."""
    doc_path = REPO_ROOT / doc_relative
    if not doc_path.exists():
        pytest.skip(f"{doc_relative} not present (may be added by later PR)")

    missing = []
    for referenced in extract_paths_from_doc(doc_path):
        rel = referenced.relative_to(REPO_ROOT).as_posix()
        if rel in ALLOWED_MISSING:
            continue
        if not referenced.exists():
            missing.append(f"{doc_relative}: references `{rel}` which does not exist")

    assert not missing, (
        "Documentation references missing files:\n" + "\n".join(missing)
    )
```
